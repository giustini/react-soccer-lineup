## Inicio Rápido

```tsx
import SoccerLineUp from 'react-soccer-lineup';

import { homeTeam, awayTeam } from '../data/teams.ts';

function App() {
  return (
    <SoccerLineUp
        color="#327d61"
        pattern="lines"
        homeTeam={homeTeam}
        awayTeam={awayTeam}
    />
  );
}
```
